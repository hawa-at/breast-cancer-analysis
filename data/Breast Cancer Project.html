encoded_data = data.copy()
therapy_cols = ['Chemotherapy', 'Hormone Therapy', 'Radio Therapy']
for col in therapy_cols:
    encoded_data[col] = encoded_data[col].map({'Yes': 1, 'No': 0})
encoded_data['Relapse Free Status'] = encoded_data['Relapse Free Status'].map({'Recurred': 1, 'Not Recurred': 0})
encoded_data['Overall Survival Status'] = encoded_data['Overall Survival Status'].map({'Living': 1, 'Deceased': 0})

features = therapy_cols + ['Tumor Size', 'Tumor Stage', 'Relapse Free Status', 'Overall Survival Status']
correlation = encoded_data[features].corr()

plt.figure(figsize=(8, 6))
sns.heatmap(correlation, annot=True, cmap="coolwarm", fmt=".2f")
plt.title("Correlation Matrix: Therapy and Clinical Outcomes")
plt.show()
